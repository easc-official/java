class trial{ 
    double length,breadth;
    
        trial(double len , double brea)
        {
            length=len;breadth=brea;
        } 
    
        trial(double value)
        {
            length=breadth=value;
        }
        double area(){
            return (length*breadth);   } 
        }
        
    class constructor {
    
    public static void main(String[] args) {
        trial t1=new trial(5.0); 
        trial t2=new trial(10.0,20.0);
        System.out.println("area of triangle 1 = "+t1.area());
    
        System.out.println("area of triangle 2 = "+t2.area());
    
    }} //end of main & class
