Number Triangle

import java.util.*;
class triangle
{
public static void main(String args[]) {

Scanner scan=new Scanner(System.in); //to get the unput from standard input
            
System.out.print(" How Many Lines You Want Print in one Triangle ? : ");
            
int count=scan.nextInt();  
            
for (int i =1;i<=count;i++)
{
 for(int j=1;j<=i;j++){

 System.out.print(" "+j);}

System.out.println();} //end of for loop
} }
