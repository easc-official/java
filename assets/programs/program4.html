import java.util.*;
import java.io.*;

public class alpha {
    
    public static void main(String[] args) {
    int i;
    Scanner sc= new Scanner(System.in);
    Scanner scan= new Scanner(System.in);

    System.out.print("\n Enter the number of Strings to be sorted : ");
    int count=sc.nextInt();
    String words[]=new String[count];
    System.out.println("\n Enter The "+count+" One by One");
    
    for(i=0;i < count;i++) {
        System.out.print(i+". ");
        words[i]=scan.nextLine(); 
    } //end of for

    Arrays.sort(words); //Sorting the array using sort() function 
    System.out.println("\n"+"The Sorted Strings Are: ");

    for(i=0;i< count;i++){ 
        System.out.println(i+". "+words[i]); 
    }

    } //main end 
}
