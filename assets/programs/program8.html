results.java
class student {
  double major,allied;
  void getmarks(double a,double b){
    major=a;
    allied=b;
  }
}

interface sports {
  double sportsmark=12.7;
  void display();
}

class results extends student implements sports {
  
  double total;
  
  void totalmark() {
    total=major+allied+sportsmark;
  }
  
  public void display() {
      System.out.println("Major ="+major);
      System.out.println("Allied ="+allied);
      System.out.println("SportsMark ="+sportsmark);  
      System.out.println("Total ="+total);
  }
  
  class mark {
    public static void main(String args[]) {
        results res=new results();
        res.getmarks(97,99);
        res.totalmark();
        res.display();
    }
  }
}
