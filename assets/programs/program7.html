balls.java
import java.io.*;
import java.awt.*;
import java.applet.*;
public class balls extends Applet implements Runnable 
{   
  int i,j,k,jj;
  Thread thread;
  public void start(){
  thread =new Thread(this);
  thread.start(); }
  public void run()
  {
    while(true){
    for(k=50;k<size().width;k+=10)
  {
  repaint();
  try{
    thread.sleep(200);
  }catch(InterruptedException e){}
  }}}
  public void stop(){
    thread.stop();
   }
  public void paint(Graphics g)
  { 
    i=k+100;
    j=k+100;
    g.setColor(Color.red);
    g.fillOval(i,175,80,80);
    g.setColor(Color.blue);
    g.fillOval(i,j+80,80,80);
    jj=k+100;
    g.setColor(Color.yellow);
    g.fillOval(i,j+180,80,80); 
   }
}
